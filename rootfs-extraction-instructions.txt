mkdir -p $HOME/rawhide-rootfs

sudo dnf5 install --installroot=$HOME/rawhide-rootfs \
  --releasever=rawhide \
  --setopt=install_weak_deps=False \
  --disablerepo="*" --enablerepo=fedora \
  --use-host-config \
  --nodocs -y @core

cd $HOME/rawhide-rootfs
sudo tar --numeric-owner --xattrs --acls -czf $HOME/Fedora-Rawhide-rootfs.tar.gz .
